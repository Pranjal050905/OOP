import java.util.Scanner;

class Publication {
    String title;
    double price;
    int copies;

    public void setData(String title, double price) {
        this.title = title;
        this.price = price;
    }

    public double saleCopy() {
        return price * copies;
    }
}

class Book extends Publication {
    String author;

    public void orderCopies(Scanner sc) {
        System.out.print("Enter author name: ");
        author = sc.nextLine();
        System.out.print("Enter number of book copies ordered: ");
        copies = sc.nextInt();
        sc.nextLine(); // consume newline
    }

    public void displayBookSale() {
        System.out.println("\nBook Details:");
        System.out.println("Title: " + title);
        System.out.println("Author: " + author);
        System.out.println("Copies Ordered: " + copies);
        System.out.println("Total Book Sale: $" + saleCopy());
    }
}

class Magazine extends Publication {
    int orderQty;
    String currentIssue;

    public void orderQty(Scanner sc) {
        System.out.print("Enter current magazine issue: ");
        currentIssue = sc.nextLine();
        System.out.print("Enter quantity ordered: ");
        orderQty = sc.nextInt();
        copies = orderQty; // assuming quantity ordered = copies
        sc.nextLine(); // consume newline
    }

    public void receiveIssue() {
        System.out.println("Received issue: " + currentIssue);
    }

    public void displayMagazineSale() {
        System.out.println("\nMagazine Details:");
        System.out.println("Title: " + title);
        System.out.println("Current Issue: " + currentIssue);
        System.out.println("Quantity Ordered: " + orderQty);
        System.out.println("Total Magazine Sale: $" + saleCopy());
    }
}

public class InteractivePublication {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Book input
        Book book = new Book();
        System.out.println("Enter details for Book:");
        System.out.print("Enter book title: ");
        String bookTitle = sc.nextLine();
        System.out.print("Enter book price: ");
        double bookPrice = sc.nextDouble();
        sc.nextLine(); // consume newline
        book.setData(bookTitle, bookPrice);
        book.orderCopies(sc);

        // Magazine input
        Magazine magazine = new Magazine();
        System.out.println("\nEnter details for Magazine:");
        System.out.print("Enter magazine title: ");
        String magTitle = sc.nextLine();
        System.out.print("Enter magazine price: ");
        double magPrice = sc.nextDouble();
        sc.nextLine(); // consume newline
        magazine.setData(magTitle, magPrice);
        magazine.orderQty(sc);

        magazine.receiveIssue(); // Call receiveIssue()

        // Display outputs
        book.displayBookSale();
        magazine.displayMagazineSale();

        double totalSale = book.saleCopy() + magazine.saleCopy();
        System.out.println("\nTotal Publication Sale: $" + totalSale);

        sc.close();
    }
}
